ls              // list segmentation 파일 보기
mkdir           // make directory 디렉토리 생성
cd              // change directory 디렉토리로 이동
cd...           // change directory ...

git init        // .git 하위 디렉토리 생성 (추적할 "폴더" 지정)
git add <파일명>  // 커밋에 단일 파일의 변경 사항을 포함 (추적할 "파일" 추가)
git status      // 파일 상태 확인
git commit      // 커밋 생성 (실제 변경사항 확정)
git log         // 현재 위치한 브랜치 커밋 내용 확인 및 식별자 부여됨
